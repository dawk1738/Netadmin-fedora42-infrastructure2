sudo useradd dawkins         Create a Samba user
sudo smbpasswd -a dawkins

sudo mkdir -p /srv/samba/share      			Create a samba share directory
sudo cp /etc/samba/smb.conf /etc/samba/smb.conf.bak
sudo nano /etc/samba/smb.conf                    Configure Samba's smb.conf 
sudo systemctl restart smb.service                Restarts Samba service to apply changes.
sudo systemctl start smb 			Starts samba services
sudo systemctl enable smb			Enables Samba to start on boot.
sudo dnf install cups				Installs printing service (CUPS).
sudo systemctl start cups			Starts CUPS service.
sudo systemctl enable cups			Enables CUPS at boot.
